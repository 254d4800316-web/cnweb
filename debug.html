Đoạn mã đã sửa + Giải thích
function addSkill(newSkill, list) {
    // Lấy MSSV của sinh viên
    const mssv = 2544800316;   // Thay bằng MSSV thật của em

    if (mssv % 2 === 0) {
        list.push(newSkill);     // Thêm vào cuối mảng (nếu MSSV chẵn)
    } else {
        list.unshift(newSkill);  // Thêm vào đầu mảng (nếu MSSV lẻ)
    }
    
    renderTable(list);           // Gọi hàm render lại bảng
}

// Giải thích lỗi:
// 1. Tên hàm sai (addCourse → addSkill)
// 2. Điều kiện if bị ngược (mssv % 2 !== 0)
// 3. Biến mssv chưa được khai báo
// 4. Comment chú thích rõ ràng hơn
